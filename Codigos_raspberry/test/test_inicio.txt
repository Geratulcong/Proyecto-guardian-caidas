[Unit]
Description=Ejecutar main.py al encender Raspberry Pi
After=network-online.target
Wants=network-online.target

[Service]
Type=simple
User=admin
WorkingDirectory=/home/admin/Desktop/Proyecto-microprocesadores1/Codigos_raspberry
ExecStart=/usr/bin/python3 /home/admin/Desktop/Proyecto-microprocesadores1/Codigos_raspberry/main.py
Restart=on-failure
RestartSec=5

[Install]
WantedBy=multi-user.target


sudo systemctl daemon-reload
sudo systemctl enable main-python.service
sudo systemctl start main-python.service
sudo systemctl status main-python.service
